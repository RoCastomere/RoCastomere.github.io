十大排序算法

我们这里统一写由小到大排序

1. 冒泡排序

复杂度：O(n^2)

算法：当相邻的两项a[i] > a[j] 且 i < j 时，交换相邻两项

int n;
int a[maxn];

for(int i = n ; i >= 1 ; i--){ // 判断结尾位置
    for(int j = 2 ; j <= i ; j++){ // 内部进行冒泡排序 
        if(a[j-1] > a[j]){
            swap(a[j-1] , a[j]);
        }
    }
}

2. 选择排序

复杂度：O(n^2)

选择剩余元素中最小or最大的，和当前位置进行交换

//maxn = 1e5 , inf = 由题意决定，这里当做1e18
int n;
int a[maxn];
for(int i = 1 ; i <= n ; i++){ //枚举需要被替换的位置
    int id = -1;
    int Min = inf;
    for(int j = i ; j <= n ; j++){ //枚举，找到被选择数字的下标
        if(Min > a[j]){
            id = j;
            Min = a[j]
        }
    }
    swap(a[i] , a[id]); //交换
}

3. 插入排序

复杂度：最好O(n)，最坏O(n^2)——倒序数组

构建已排序序列，在已排序中从后向前，找到位置并插入数据

int n;
int a[maxn];
for(int i = 1 ; i <= n ; i++){
    int id = i; 寻找的插入位置
    for(int j = i-1 ; j >= 1 ; j--){
        if(a[i] >= a[j]){
        // 找到一个a[j] <= a[j]，将a[i]放到它后面，退出循环
            a[j+1] = a[i];
            break;
        } else {
            a[j+1] = a[j]; // 比a[i] 大的向后挪
        }
    }

}

4. 希尔排序

时间复杂度：最坏O(n^2),最好O(n)

通过确定K个增量t1，t2…tk,对按增量划分的 子序列 进行插入排序

(取出所有间隔tk-1的项，进行插入排序)

int n;
int a[maxn];

for(int gap = n/2 ; gap > 0 ; gap /= 2){
    //增量是gap，每次变为之前的 1/2
    for(int i = gap ; i <= len ; i++){
        int x = a[i] , id = i;
        for(int j = i ; j - gap >= 0 && x <= a[j-gap] ; j -= gap , id = j){
            a[j] = a[j-gap];
        }
        a[j] = x;
    }
}

5. 归并排序

时间复杂度O(nlogn)

每次对一段区间 一分为二，直到为区间长度为 1，每次向上有序合并区间 

int n;
int a[maxn];
void MergeSort(int l, int r, int len){ //初始传入 (1 , n , n) 
    if(len <= 1) return;
    int mid = (l + r) >> 1;
    MergeSort(l, mid, mid-l+1);
    MergeSort(mid+1, r, r-mid);
    int t[maxn] , cnt= 0;
    int dl = mid-l+1 , dr = r-mid;
    whle(dl && dr){
        int c1 = a[mid-dl+1] , c2 = a[r-dr+1];
        if(c1 <= c2){
            t[++cnt] = c[mid-dl+1];
            dl--;
        } else {
            t[++cnt] = c[r-dr+1];
            dr--;
        }
    }
    while(dl){
        t[++cnt] = c[mid-dl+1];
        dl--;
    }
    while(dr){
        t[++cnt] = c[r-dr+1];
        dr--;
    }
    for(int i = l , j = 1; i <= r ; i++ , j++){
        a[i] = t[j];
    }
}

6. 快速排序

时间复杂度：最慢O(n^2),最快O(nlogn)

一个区间，选择基准值，比它小的放前面，大的放后面

然后递归实现，减小这种区间的大小

void quicksort(int arr[] , int i , int j){
    if(i > j)    return;
    int mid = arr[j]; // 选择基准值
    int l = i , r = j - 1; //基准值之外的区间
    while(l < r){
        while(arr[l] < mid && l < r) l++; // 左边区间比它小的忽略
        while(arr[r] >= mid && l < r) r--; // 右边比它大的忽略
        swap(arr[l] , arr[r]); // 找到的不能忽略的两个值，交换
    }
    if(arr[l] >= arr[j])    swap(arr[l] , arr[j]); // 看a[l] 这个位置和基准值比大小，并且交换
    else l++;
    //下层递归
    if(l)    quicksort(arr , i , l - 1);
    quicksort(arr , l + 1 , j);
}

7. 堆排序

时间复杂度O(nlogn)

int n , len;
int a[maxn];
void heapify(int i){
    int rt = i, ls = i * 2 + 1, rs = i * 2 + 2;
    if(ls < len && a[ls] > a[rt]){
        rt = rt;
    }
    if(rs < len && a[rs] > a[rt]){
        rs = rt;
    }
    if(rt != i){
        swap(a[i] , a[rt]);
        heapify(rt);
    }
}

void Build(){
    for(int i = n/2 ; i>= 0 ; i--){
        heapify(i); // 堆调整
    }
}


void heapSort(){
    Build();
    len = n;
    for(int i = n-1 ; i > 0 ; i--){
        swap(a[0] , a[i]);
        len--;
        heapify(0);
    }
}

8. 计数排序

时间复杂度O(n)

一种利用额外空间vis[]，对已知数组每个数 计数的方式，然后从小到大，线性输出

要求排序有范围，让vis可以进行标记，如果数字超出vis标记，用map or 哈希方法，对应计数

int n;
int vis[maxn];
int a[maxn];
for(int i = 1 ; i <= n ; i++){
    vis[a[i]]++;
}
int ans[maxn] , tot = 0;
for(int i = 1 ; i <= maxn ; i++){
    while(vis[i]--){
        ans[++tot] = i;
    }
}

9. 桶排序

时间复杂度：最坏O(n^2),最好O(n)

假设数据服从均匀分布，选一个定量数组当空桶，每个桶内排序，最后扫一遍数组

实现方法：

用d进制分桶，i次桶排排倒数第i位，对第i+1位桶排，每一个桶都是后i位有序的

10. 基数排序

复杂度：O(nk) , k进制

基数排序是按照低位先排序，然后收集；

再按照高位排序，然后再收集；

依次类推，直到最高位。有时候有些属性是有优先级顺序的，先按低优先级排序，再按高优先级排序。最后的次序就是高优先级高的在前，高优先级相同的低优先级高的在前。
